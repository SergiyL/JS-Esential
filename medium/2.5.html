Write a JavaScript function to find the first not repeated
character. For i.e. input 'abacddbec', should return 'e' 


            function not_repeat(str) {
                var duplicates = '';
                for (var i = 0; i <= str.length-1; i++) {
                    if (str.substring(i+1, str.length).includes(str.charAt(i)) || duplicates.includes(str.charAt(i))) {
                        duplicates += str.charAt(i);
                    } else {
                        return str.charAt(i);
                    }
                }
            }
            
            not_repeat(str);
