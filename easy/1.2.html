Write a function that returns min and max numbers.
For i.e. input [1,92,-5,25] should return {max:92, min:-5} 


            function min_max(arr) {
                var max = -Infinity;
                var min = Infinity;
                for (i of arr) {
                    if (min > i) {
                        min = i;
                    }
                    if (max < i) {
                        max = i;
                    }
                }
                return "max:" + max + ", min:" + min;
            }
            
            min_max(arr);
