                function validator(input) {
                    var count = 0;
                    var obj = {name: "Vasya", age: 20, iq: 80};
                    var validkeys = Object.keys(obj);
                    for (var i = 0; i <= input.length-1; i++) {
                        if (validkeys.includes(input[i])) {
                            count ++;
                        } else {
                            count --;
                        }
                    }
                
                    if (count == input.length) {
                        return true;
                    } else {
                        return false;
