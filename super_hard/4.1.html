            function bubble(arr) {
                var temp = 0;
                for (var i = 0; i < arr.length; i++) {
                    for (var j = i+1; j < arr.length; j++) {
                        if (arr[i] > arr[j]) {
                            temp = arr[i];
                            console.log(temp);
                            arr[i] = arr[j];
                            arr[j] = temp;
                        }
                    }
                }
                return arr;
            }
